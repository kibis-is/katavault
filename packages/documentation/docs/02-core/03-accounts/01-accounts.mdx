# accounts

> Gets a list of the accounts within the vault.

:::note

Authentication is **REQUIRED**.

:::

<SectionBreak />

## Usage

<Tabs
  defaultValue="main"
  values={[
    { label: 'main.ts', value: 'main' },
  ]}>
  <TabItem value="main">

    ```typescript
    import { Voi } from '@kibisis/chains';
    import { type Account, createKatavault } from '@kibisis/katavault-core';

    const katavault = await createKatavault({
      chains: [Voi],
    });
    let accounts: Account[];

    await katavault.authenticate();

    accounts = await katavault.accounts();

    console.log(accounts);
    /*
    [
      {
        key: 'FJ4XvAApLhdvYFd2umNSsb87KE6rpuXQcfptb8WZ9Feo',
        name: 'Account-1',
      },
      {
        key: 'CZepAkxcz4KEruVyrURwGFh8SEkKLFL8nASuXmS2Zu3a',
        name: 'Account-2',
      },
    ]
    */
    ```

  </TabItem>
</Tabs>
