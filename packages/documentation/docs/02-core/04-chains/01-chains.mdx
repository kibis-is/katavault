# chains

> Gets the supported chains.

<SectionBreak />

## Usage

<Tabs
  defaultValue="main"
  values={[
    { label: 'main.ts', value: 'main' },
  ]}>
  <TabItem value="main">

    ```typescript
    import { type Chain, Voi } from '@kibisis/chains';
    import { createKatavault } from '@kibisis/katavault-core';

    const katavault = await createKatavault({
      chains: [Voi],
    });
    const chains = katavault.chains();

    console.log(chains);
    /*
    [
      {
        displayName: 'Voi Network',
        namespace: 'avm',
        reference: 'r20fSQI8gWe_kFZziNonSPCXLwcQmH_n',
        ...
      },
    ]
    */
    ```

  </TabItem>
</Tabs>
