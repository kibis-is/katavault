# useAddChain

> Hook to add a chain to the vault.

<SectionBreak />

## Usage

<Tabs
  defaultValue="chains"
  values={[
    { label: 'Chains.tsx', value: 'chains' },
    { label: 'App.tsx', value: 'app' },
    { label: 'config.ts', value: 'config' },
  ]}>
  <TabItem value="chains">

    ```typescript
    import { Algorand, Chain } from '@kibisis/chains';
    import { useAddChain, useChains } from '@kibisis/katavault-react';
    import { useCallback } from 'react';

    function Chains() {
      const chains = useChains();
      const addChain = useAddChain();
      const handleOnAddAlgorandClick = useCallback(() => {
        addChain(Algorand, {
          onError: (error) => {
            console.error(error);
          },
          onSuccess: (result: Chain) => {
            console.log(`added new chain ${result.chainID()}`);
          },
        });
      }, [addChain]);

      return (
        <div>
          {chains.map((chain) => (
            <div>
              <p>Name: {chain.displayName()}</p>
              <p>Chain ID: {chain.chainID()}</p>
            </div>
          )}

          <button onClick={handleOnAddAlgorandClick}>Add Algorand</button>
        </div>
      );
    }
    ```

  </TabItem>
  <TabItem value="app">

    ```typescript
    import { KatavaultProvider } from '@kibisis/katavault-react';
    import { Chains } from './Chains';
    import config from './config';

    function App() {
      return (
        <KatavaultProvider config={config}>
          <Chains />
        </KatavaultProvider>
      );
    }
    ```

  </TabItem>


## Returns

`(params?: ChainConstructor, options?: { onError, onSuccess }) => void`

The function can be called with callbacks that are invoked on error/success.

#### options.onError

Type: `(error: BaseError, params?: Chain) => void | Promise<void>`

This will be invoked if there was an error.

#### options.onSuccess

Type: `(result: Chain, params?: Chain) => void | Promise<void>`

This will be invoked if the chain was added successfully.
